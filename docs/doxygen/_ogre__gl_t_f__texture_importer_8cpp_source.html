78 bool textureImporter::isHardwareGammaEnabled() const
79 {
80  const auto renderSystem = Ogre::Root::getSingleton().getRenderSystem();
81 
82  //don't bother to check render targets for gamma if render system doesn't have the option enabled
83  try
84  {
85  const auto renderSystemGammaConversionConfigOption = renderSystem->getConfigOptions().at("sRGB Gamma Conversion");
86  if(renderSystemGammaConversionConfigOption.currentValue != "Yes") return false;
87  }
88  catch(const std::exception& e)
89  {
90  OgreLog("It appears that render system doesn't know if it uses srgb gamma conversion? How is that possible? We're going to check the render targets "
91  "anyway");
92  }
93 
94  //Check if any render target has the option, and if so, return true
95  auto renderTargetIt = renderSystem->getRenderTargetIterator();
96  do
97  {
98  if(renderTargetIt.current()->second->isHardwareGammaEnabled()) return true;
99 
100  renderTargetIt.moveNext();
101  } while(renderTargetIt.current() != renderTargetIt.end());
102 
103  return false;
104 }
