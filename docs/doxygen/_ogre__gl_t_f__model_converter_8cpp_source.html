14 Ogre::VertexBufferPackedVec modelConverter::constructVertexBuffer(const std::vector<vertexBufferPart>& parts) const
15 {
16  Ogre::VertexElement2Vec vertexElements;
17 
18  size_t stride { 0 }, strideInElements { 0 };
19  size_t vertexCount { 0 }, previousVertexCount { 0 };
20 
21  for(const auto& part : parts)
22  {
23  vertexElements.emplace_back(part.type, part.semantic);
24  strideInElements += part.perVertex;
25  stride += part.buffer->elementSize() * part.perVertex;
26  vertexCount = part.vertexCount;
27 
28  //Sanity check
29  if(previousVertexCount != 0)
30  {
31  if(vertexCount != previousVertexCount) throw std::runtime_error("Part of vertex buffer for the same primitive have different vertex counts!");
32  }
33  else
34  previousVertexCount = vertexCount;
35  }
36 
37  OgreLog("There will be " + std::to_string(vertexCount) + " vertices with a stride of " + std::to_string(stride) + " bytes");
38 
39  geometryBuffer<float> finalBuffer(vertexCount * strideInElements);
40  size_t bytesWrittenInCurrentStride { 0 };
41  for(size_t vertexIndex = 0; vertexIndex < vertexCount; ++vertexIndex)
42  {
43  bytesWrittenInCurrentStride = 0;
44  for(const auto& part : parts)
45  {
46  memcpy(finalBuffer.dataAddress() + (bytesWrittenInCurrentStride + vertexIndex * stride),
47  (part.buffer->dataAddress() + (vertexIndex * part.getPartStride())),
48  part.getPartStride());
49  bytesWrittenInCurrentStride += part.getPartStride();
50  }
51  }
52 
53  Ogre::VertexBufferPackedVec vec;
54  auto vertexBuffer = getVaoManager()->createVertexBuffer(vertexElements, vertexCount, Ogre::BT_IMMUTABLE, finalBuffer.data(), false);
55 
56  vec.push_back(vertexBuffer);
57  return vec;
58 }

203 ModelInformation::ModelTransform modelConverter::getTransform()
204 {
205  ModelInformation::ModelTransform trans;
206  std::array<float, 3> translation { 0 }, scale { 0 };
207  std::array<float, 4> rotation { 0 };
208  std::array<float, 4 * 4> local_matrix { 0 };
209  bool set = false;
210 
211  // Just get the first one - not sure if there can be more for a model but doubt it
212  const auto& nodes = (model.defaultScene != 0 ? model.nodes[model.scenes[model.defaultScene].nodes[0]] : model.nodes[0]);
213  if(!nodes.translation.empty())
214  {
215  internal_utils::container_double_to_float(nodes.translation, translation);
216  trans.position = Ogre::Vector3 { translation.data() };
217  set = true;
218  }
219  if(!nodes.scale.empty())
220  {
221  internal_utils::container_double_to_float(nodes.scale, scale);
222  trans.scale = Ogre::Vector3 { scale.data() };
223  set = true;
224  }
225  if(!nodes.rotation.empty())
226  {
227  internal_utils::container_double_to_float(nodes.rotation, rotation);
228  trans.orientation = Ogre::Quaternion { rotation[3], rotation[0], rotation[1], rotation[2] };
229  set = true;
230  }
231 
232  if(!set && !nodes.matrix.empty())
233  {
234  internal_utils::container_double_to_float(nodes.matrix, local_matrix);
235  Ogre::Matrix4 transform_matrix { local_matrix.data() };
236 
237  transform_matrix.transpose().decomposition(trans.position, trans.scale, trans.orientation);
238  }
239 
240  return trans;
241 }

249 Ogre::IndexBufferPacked* modelConverter::extractIndexBuffer(int accessorID) const
250 {
251  OgreLog("Extracting index buffer");
252  const auto& accessor = model.accessors[accessorID];
253  const auto& bufferView = model.bufferViews[accessor.bufferView];
254  auto& buffer = model.buffers[bufferView.buffer];
255  const auto byteStride = accessor.ByteStride(bufferView);
256  const auto indexCount = accessor.count;
257  Ogre::IndexBufferPacked::IndexType type;
258 
259  if(byteStride < 0) throw std::runtime_error("Can't get valid bytestride from accessor and bufferview. Loading data not possible");
260 
261  auto convertTo16Bit { false };
262  switch(accessor.componentType)
263  {
264  default: throw std::runtime_error("Unrecognized index data format");
265  case TINYGLTF_COMPONENT_TYPE_BYTE:
266  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_BYTE: convertTo16Bit = true;
267  case TINYGLTF_COMPONENT_TYPE_SHORT:
268  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_SHORT:
269  {
270  type = Ogre::IndexBufferPacked::IT_16BIT;
271  auto geomBuffer = geometryBuffer<Ogre::uint16>(indexCount);
272  if(convertTo16Bit)
273  loadIndexBuffer(geomBuffer.data(), buffer.data.data(), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
274  else
275  loadIndexBuffer(geomBuffer.data(),
276  reinterpret_cast<Ogre::uint16*>(buffer.data.data()),
277  indexCount,
278  bufferView.byteOffset + accessor.byteOffset,
279  byteStride);
280  return getVaoManager()->createIndexBuffer(type, indexCount, Ogre::BT_IMMUTABLE, geomBuffer.dataAddress(), false);
281  }
282  case TINYGLTF_COMPONENT_TYPE_INT:;
283  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_INT:
284  {
285  type = Ogre::IndexBufferPacked::IT_32BIT;
286  auto geomBuffer = geometryBuffer<Ogre::uint32>(indexCount);
287  loadIndexBuffer(
288  geomBuffer.data(), reinterpret_cast<Ogre::uint32*>(buffer.data.data()), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
289  return getVaoManager()->createIndexBuffer(type, indexCount, Ogre::BT_IMMUTABLE, geomBuffer.dataAddress(), false);
290  }
291  }
292 }

318 vertexBufferPart modelConverter::extractVertexBuffer(const std::pair<std::string, int>& attribute, Ogre::Aabb& boundingBox) const
319 {
320  const auto elementScemantic = getVertexElementScemantic(attribute.first);
321  const auto& accessor = model.accessors[attribute.second];
322  const auto& bufferView = model.bufferViews[accessor.bufferView];
323  const auto& buffer = model.buffers[bufferView.buffer];
324  const auto vertexBufferByteLen = bufferView.byteLength;
325  const auto numberOfElementPerVertex = getVertexBufferElementsPerVertexCount(accessor.type);
326  const auto elementOffsetInBuffer = bufferView.byteOffset + accessor.byteOffset;
327  size_t bufferLenghtInBufferBasicType { 0 };
328 
329  std::unique_ptr<geometryBuffer_base> geomBuffer { nullptr };
330  Ogre::VertexElementType elementType {};
331 
332  switch(accessor.componentType)
333  {
334  case TINYGLTF_COMPONENT_TYPE_DOUBLE: throw std::runtime_error("Double pressision not implemented!");
335  case TINYGLTF_COMPONENT_TYPE_FLOAT:
336  bufferLenghtInBufferBasicType = (vertexBufferByteLen / sizeof(float));
337  geomBuffer = std::make_unique<geometryBuffer<float>>(bufferLenghtInBufferBasicType);
338  if(numberOfElementPerVertex == 2) elementType = Ogre::VET_FLOAT2;
339  if(numberOfElementPerVertex == 3) elementType = Ogre::VET_FLOAT3;
340  if(numberOfElementPerVertex == 4) elementType = Ogre::VET_FLOAT4;
341  break;
342  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_SHORT:
343  bufferLenghtInBufferBasicType = (vertexBufferByteLen / sizeof(unsigned short));
344  geomBuffer = std::make_unique<geometryBuffer<unsigned short>>(bufferLenghtInBufferBasicType);
345  if(numberOfElementPerVertex == 2) elementType = Ogre::VET_USHORT2;
346  if(numberOfElementPerVertex == 4) elementType = Ogre::VET_USHORT4;
347  break;
348  default: throw std::runtime_error("Unrecognized vertex buffer coponent type");
349  }
350 
351  //if(bufferView.byteStride == 0)
352  // OgreLog("Vertex buffer is 'tightly packed'");
353 
354  const auto byteStride = accessor.ByteStride(bufferView);
355  const auto vertexCount = accessor.count;
356  const auto vertexElementLenghtInBytes = numberOfElementPerVertex * geomBuffer->elementSize();
357 
358  if(byteStride < 0) throw std::runtime_error("Can't get valid bytestride from accessor and bufferview. Loading data not possible");
359 
360  //OgreLog("A vertex element on this buffer is " + std::to_string(vertexElementLenghtInBytes) + " bytes long");
361  for(size_t vertexIndex = 0; vertexIndex < vertexCount; vertexIndex++)
362  {
363  const auto destOffset = vertexIndex * vertexElementLenghtInBytes;
364  const auto sourceOffset = elementOffsetInBuffer + vertexIndex * byteStride;
365 
366  memcpy((geomBuffer->dataAddress() + destOffset), (buffer.data.data() + sourceOffset), vertexElementLenghtInBytes);
367  }
368 
369  //Update the bounding sizes once, when vertex positions has been read.
370  if(elementScemantic == Ogre::VES_POSITION)
371  {
372  //Convert to float and load into Ogre::Vector3 objects
373  std::array<float, 3> floatVector {};
374  internal_utils::container_double_to_float(accessor.minValues, floatVector);
375  const Ogre::Vector3 minBounds { floatVector.data() };
376  internal_utils::container_double_to_float(accessor.maxValues, floatVector);
377  const Ogre::Vector3 maxBounds { floatVector.data() };
378 
379  OgreLog("Updating bounding box size: ");
380  OgreLog("Setting Min size: " + std::to_string(minBounds.x) + " " + std::to_string(minBounds.y) + " " + std::to_string(minBounds.z));
381  OgreLog("Setting Max size: " + std::to_string(maxBounds.x) + " " + std::to_string(maxBounds.y) + " " + std::to_string(maxBounds.z));
382  boundingBox.merge(Ogre::Aabb(minBounds, maxBounds));
383  }
384 
385  //geometryBuffer->_debugContentToLog();
386  return { std::move(geomBuffer), elementType, elementScemantic, vertexCount, numberOfElementPerVertex };
387 }
