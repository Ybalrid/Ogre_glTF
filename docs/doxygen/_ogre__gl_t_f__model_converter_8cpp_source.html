14 Ogre::VertexBufferPackedVec modelConverter::constructVertexBuffer(const std::vector<vertexBufferPart>& parts) const
15 {
16  Ogre::VertexElement2Vec vertexElements;
17 
18  size_t stride { 0 }, strideInElements { 0 };
19  size_t vertexCount { 0 }, previousVertexCount { 0 };
20 
21  for(const auto& part : parts)
22  {
23  vertexElements.emplace_back(part.type, part.semantic);
24  strideInElements += part.perVertex;
25  stride += part.buffer->elementSize() * part.perVertex;
26  vertexCount = part.vertexCount;
27 
28  //Sanity check
29  if(previousVertexCount != 0)
30  {
31  if(vertexCount != previousVertexCount) throw std::runtime_error("Part of vertex buffer for the same primitive have different vertex counts!");
32  }
33  else
34  previousVertexCount = vertexCount;
35  }
36 
37  OgreLog("There will be " + std::to_string(vertexCount) + " vertices with a stride of " + std::to_string(stride) + " bytes");
38 
39  geometryBuffer<float> finalBuffer(vertexCount * strideInElements);
40  size_t bytesWrittenInCurrentStride { 0 };
41  for(size_t vertexIndex = 0; vertexIndex < vertexCount; ++vertexIndex)
42  {
43  bytesWrittenInCurrentStride = 0;
44  for(const auto& part : parts)
45  {
46  memcpy(finalBuffer.dataAddress() + (bytesWrittenInCurrentStride + vertexIndex * stride),
47  (part.buffer->dataAddress() + (vertexIndex * part.getPartStride())),
48  part.getPartStride());
49  bytesWrittenInCurrentStride += part.getPartStride();
50  }
51  }
52 
53  Ogre::VertexBufferPackedVec vec;
54  auto vertexBuffer = getVaoManager()->createVertexBuffer(vertexElements, vertexCount, Ogre::BT_IMMUTABLE, finalBuffer.data(), false);
55 
56  vec.push_back(vertexBuffer);
57  return vec;
58 }

209 Ogre::IndexBufferPacked* modelConverter::extractIndexBuffer(int accessorID) const
210 {
211  OgreLog("Extracting index buffer");
212  const auto& accessor = model.accessors[accessorID];
213  const auto& bufferView = model.bufferViews[accessor.bufferView];
214  auto& buffer = model.buffers[bufferView.buffer];
215  const auto byteStride = accessor.ByteStride(bufferView);
216  const auto indexCount = accessor.count;
217  Ogre::IndexBufferPacked::IndexType type;
218 
219  if(byteStride < 0) throw std::runtime_error("Can't get valid bytestride from accessor and bufferview. Loading data not possible");
220 
221  auto convertTo16Bit { false };
222  switch(accessor.componentType)
223  {
224  default: throw std::runtime_error("Unrecognized index data format");
225  case TINYGLTF_COMPONENT_TYPE_BYTE:
226  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_BYTE: convertTo16Bit = true;
227  case TINYGLTF_COMPONENT_TYPE_SHORT:
228  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_SHORT:
229  {
230  type = Ogre::IndexBufferPacked::IT_16BIT;
231  auto geomBuffer = geometryBuffer<Ogre::uint16>(indexCount);
232  if(convertTo16Bit)
233  loadIndexBuffer(geomBuffer.data(), buffer.data.data(), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
234  else
235  loadIndexBuffer(geomBuffer.data(),
236  reinterpret_cast<Ogre::uint16*>(buffer.data.data()),
237  indexCount,
238  bufferView.byteOffset + accessor.byteOffset,
239  byteStride);
240  return getVaoManager()->createIndexBuffer(type, indexCount, Ogre::BT_IMMUTABLE, geomBuffer.dataAddress(), false);
241  }
242  case TINYGLTF_COMPONENT_TYPE_INT:;
243  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_INT:
244  {
245  type = Ogre::IndexBufferPacked::IT_32BIT;
246  auto geomBuffer = geometryBuffer<Ogre::uint32>(indexCount);
247  loadIndexBuffer(
248  geomBuffer.data(), reinterpret_cast<Ogre::uint32*>(buffer.data.data()), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
249  return getVaoManager()->createIndexBuffer(type, indexCount, Ogre::BT_IMMUTABLE, geomBuffer.dataAddress(), false);
250  }
251  }
252 }

278 vertexBufferPart modelConverter::extractVertexBuffer(const std::pair<std::string, int>& attribute, Ogre::Aabb& boundingBox) const
279 {
280  const auto elementScemantic = getVertexElementScemantic(attribute.first);
281  const auto& accessor = model.accessors[attribute.second];
282  const auto& bufferView = model.bufferViews[accessor.bufferView];
283  const auto& buffer = model.buffers[bufferView.buffer];
284  const auto vertexBufferByteLen = bufferView.byteLength;
285  const auto numberOfElementPerVertex = getVertexBufferElementsPerVertexCount(accessor.type);
286  const auto elementOffsetInBuffer = bufferView.byteOffset + accessor.byteOffset;
287  size_t bufferLenghtInBufferBasicType { 0 };
288 
289  std::unique_ptr<geometryBuffer_base> geomBuffer { nullptr };
290  Ogre::VertexElementType elementType {};
291 
292  switch(accessor.componentType)
293  {
294  case TINYGLTF_COMPONENT_TYPE_DOUBLE: throw std::runtime_error("Double pressision not implemented!");
295  case TINYGLTF_COMPONENT_TYPE_FLOAT:
296  bufferLenghtInBufferBasicType = (vertexBufferByteLen / sizeof(float));
297  geomBuffer = std::make_unique<geometryBuffer<float>>(bufferLenghtInBufferBasicType);
298  if(numberOfElementPerVertex == 2) elementType = Ogre::VET_FLOAT2;
299  if(numberOfElementPerVertex == 3) elementType = Ogre::VET_FLOAT3;
300  if(numberOfElementPerVertex == 4) elementType = Ogre::VET_FLOAT4;
301  break;
302  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_SHORT:
303  bufferLenghtInBufferBasicType = (vertexBufferByteLen / sizeof(unsigned short));
304  geomBuffer = std::make_unique<geometryBuffer<unsigned short>>(bufferLenghtInBufferBasicType);
305  if(numberOfElementPerVertex == 2) elementType = Ogre::VET_USHORT2;
306  if(numberOfElementPerVertex == 4) elementType = Ogre::VET_USHORT4;
307  break;
308  default: throw std::runtime_error("Unrecognized vertex buffer coponent type");
309  }
310 
311  //if(bufferView.byteStride == 0)
312  // OgreLog("Vertex buffer is 'tightly packed'");
313 
314  const auto byteStride = accessor.ByteStride(bufferView);
315  const auto vertexCount = accessor.count;
316  const auto vertexElementLenghtInBytes = numberOfElementPerVertex * geomBuffer->elementSize();
317 
318  if(byteStride < 0) throw std::runtime_error("Can't get valid bytestride from accessor and bufferview. Loading data not possible");
319 
320  //OgreLog("A vertex element on this buffer is " + std::to_string(vertexElementLenghtInBytes) + " bytes long");
321  for(size_t vertexIndex = 0; vertexIndex < vertexCount; vertexIndex++)
322  {
323  const auto destOffset = vertexIndex * vertexElementLenghtInBytes;
324  const auto sourceOffset = elementOffsetInBuffer + vertexIndex * byteStride;
325 
326  memcpy((geomBuffer->dataAddress() + destOffset), (buffer.data.data() + sourceOffset), vertexElementLenghtInBytes);
327  }
328 
329  //Update the bounding sizes once, when vertex positions has been read.
330  if(elementScemantic == Ogre::VES_POSITION)
331  {
332  //Convert to float and load into Ogre::Vector3 objects
333  std::array<float, 3> floatVector {};
334  internal_utils::container_double_to_float(accessor.minValues, floatVector);
335  const Ogre::Vector3 minBounds { floatVector.data() };
336  internal_utils::container_double_to_float(accessor.maxValues, floatVector);
337  const Ogre::Vector3 maxBounds { floatVector.data() };
338 
339  OgreLog("Updating bounding box size: ");
340  OgreLog("Setting Min size: " + std::to_string(minBounds.x) + " " + std::to_string(minBounds.y) + " " + std::to_string(minBounds.z));
341  OgreLog("Setting Max size: " + std::to_string(maxBounds.x) + " " + std::to_string(maxBounds.y) + " " + std::to_string(maxBounds.z));
342  boundingBox.setExtents(minBounds, maxBounds);
343  }
344 
345  //geometryBuffer->_debugContentToLog();
346  return { std::move(geomBuffer), elementType, elementScemantic, vertexCount, numberOfElementPerVertex };
347 }
