120  FileType detectType(const std::string& path) const
121  {
122  //Quickly open the file as binary and check if there's the gltf binary magic number
123  {
124  auto probe = std::ifstream(path, std::ios_base::binary);
125  if(!probe) throw std::runtime_error("Could not open " + path);
126 
127  std::array<char, 5> buffer {};
128  for(size_t i { 0 }; i < 4; ++i) probe >> buffer[i];
129  buffer[4] = 0;
130 
131  if(std::string("glTF") == std::string(buffer.data()))
132  {
133  //OgreLog("Detected binary file thanks to the magic number at the start!");
134  return FileType::Binary;
135  }
136  }
137 
138  //If we don't have any better, check the file extension.
139  auto extension = path.substr(path.find_last_of('.') + 1);
140  std::transform(std::begin(extension), std::end(extension), std::begin(extension), [](char c) { return char(tolower(int(c))); });
141  if(extension == "gltf") return FileType::Ascii;
142  if(extension == "glb") return FileType::Binary;
143 
144  return FileType::Unknown;
145  }
146 
148  bool loadInto(loaderAdapter& adapter, const std::string& path)
149  {
150  switch(detectType(path))
151  {
152  default:
153  case FileType::Unknown: return false;
154  case FileType::Ascii:
155  //OgreLog("Detected ascii file type");
156  return loader.LoadASCIIFromFile(&adapter.pimpl->model, &adapter.pimpl->error, &adapter.pimpl->warnings, path);
157  case FileType::Binary:
158  //OgreLog("Deteted binary file type");
159  return loader.LoadBinaryFromFile(&adapter.pimpl->model, &adapter.pimpl->error, &adapter.pimpl->warnings, path);
160  }
161  }
